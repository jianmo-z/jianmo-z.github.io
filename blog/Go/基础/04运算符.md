# 04运算符
```
package main

import "fmt"

/*
 *  算术运算符、关系运算符、逻辑运算符、位运算符、赋值运算符、其他运算符
 *
 *  算术运算符：+ - * / % ++ --
 *  关系运算符：== != > < >= <=
 *  逻辑运算符：&& || !
 *  位运算符：& | ^ << >>
 *  赋值运算符：= += -= *= /= %= <<= >>= &= ^= |=
 *  其他运算符：&(返回变量存储地址) *(指针变量)
 */
func main(){
	var a = "hello" // 可以自动判断变量类型，所以string类型名可以省略掉
	var b = "world"
	c := a + b
	fmt.Printf("c = %s, len(c) = %d", c, len(c))
}

```
